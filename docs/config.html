set

adds a configuration

 
    set : (key, value) ->
      return if @cache[key] is value
      
      @cache[key] = value
      
      update = {}
      update[key] = value
      
      is_silent = key.charAt(0) is '_'
      @hoodie.store.update @type, @id, update, silent: is_silent
