    hoodieAppUrl = 'http://myapp.hood.ie';

    hoodie.my.account.signUp('joe@example.com', 'secret')
      .done( function(user) { } )

    // sign in
    hoodie.my.account.signIn('joe@example.com', 'secret')
      .done( function(user) { } ) 
      .fail( function(err)  { } )
      
    // sign out
    hoodie.my.account.signOut()
      .done( function(user) { } ) 
      .fail( function(err)  { } )

    // forgot password (coming soon)
    hoodie.my.account.forgotPassword('joe@example.com')
      .done( function(user) { } ) 
      .fail( function(err)  { } )
      
    //
    // Store
    //
    
    // helper to generate uuids
    uuid = hoodie.my.localStore.uuid(length)
    
    // create a new object
    type = 'couch'
    attributes = {color: "red"}
    hoodie.my.localStore.create(type, attributes )
      .done ( function(newObject) { } )
      .fail ( function(err)        { } )
      
    // save an object
    type = 'couch'
    id   = 'abc4567'
    attributes  = {color: "red", name: "relax"}
    hoodie.my.localStore.save( type, id, attributes )
      .done ( function(object) { } )
      .fail ( function(err)        { } )
      
    // update an existing object
    type = 'couch'
    id   = 'abc4567'
    update = {size: 2}
    hoodie.my.localStore.update( type, id, update )
      .done ( function(updatedObject) { } )
      .fail ( function(err)            { } )
    
    // load one object
    type = 'couch'
    id   = 'abc4567'
    hoodie.my.localStore.load( type, id )
      .done ( function(object) { } )
      .fail ( function(err)    { } )
      
    // Load all objects
    hoodie.my.localStore.loadAll()
      .done ( function(objects) { } )
      .fail ( function(err)     { } )
      
    // Load all objects from one type
    type = 'couch'
    hoodie.my.localStore.loadAll( type )
      .done ( function(objects) { } )
      .fail ( function(err)     { } )
      
    // delete an existing object
    type = 'couch'
    id   = 'abc4567'
    hoodie.my.localStore.delete( type, id )
      .done ( function(deletedObject) { } )
      .fail ( function(err)            { } )
      
    //
    // Synchronization
    //
    // When signed in, local changes do get synched automatically.
    // You can subscribe to remote updates
    // 
    
    // new doc created
    hoodie.my.remoteStore.on( 'create', function( type, id, createdObject) { } )

    // existing doc updated
    hoodie.my.remoteStore.on( 'update', function( type, id, updatedObject) { } )

    // doc deleted
    hoodie.my.remoteStore.on( 'delete', function( type, id, deletedObject) { } )

    // any of above events
    hoodie.my.remoteStore.on( 'change', function( type, id, changedObject) { } )

    // all listeners can be filtered by type
    hoodie.my.remoteStore.on( "create:couch", function( id, createdObject) { } )
    hoodie.my.remoteStore.on( "update:couch", function( id, updatedObject)  { } )
    hoodie.my.remoteStore.on( "delete:couch", function( id, deletedObject) { } )
    hoodie.my.remoteStore.on( "change:couch", function( id, changedObject) { } )
